# Pseudocode for Number Pattern

BEGIN

    // Start of the main program
    SET i to 1

    WHILE i is less than or equal to 5 DO
        SET j to 1
        
        WHILE j is less than or equal to i DO
            PRINT j
            INCREMENT j by 1
        END WHILE
        
        PRINT a new line
        INCREMENT i by 1
    END WHILE

    RETURN 0 // End of the program

END
